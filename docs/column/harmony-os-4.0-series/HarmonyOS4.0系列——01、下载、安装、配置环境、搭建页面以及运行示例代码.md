```ts
@Entry
// 程序入口
@Component
// 组件

// 结构体，语法格式 struct {}
struct Index {
  @State message: string = 'Hello World'

  build() {
    // 界面内容放在build中
    Row() {
      // 行有多高
      Column() {
        // 列有多长
        Text(this.message) 
          .fontSize(50)
          .fontWeight(FontWeight.Bold)
      }
      .width('100%')
    }
    .height('100%')
  }
}
```

此段代码就是让界面撑满整个屏幕并对`Hello World`设置字体大小以及粗体
